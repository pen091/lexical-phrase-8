#include <stdio.h>

typedef unsigned int uint;

struct Point {
    int x;
    int y;
};

int main(void) {
    const uint MAX = 5;
    struct Point p = {0, 0};

    for (uint i = 0; i < MAX; i++) {
        p.x += i;
        if (p.x > 5)
            break;
    }

    return 0;
